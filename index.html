// Headers
Authorization: Bearer {sessionToken}
X-Device-ID: {deviceId}
Content-Type: application/json

// Request body
{
  "url": "https://linkvertise.com/...",
  "forceRefresh": false
}

// Response (success)
{
  "result": "https://url-tujuan-asli.com/..."
}

Contoh implementasi (Node.js)

// utils/bypass.js
const crypto = require("crypto");

const bypass = async (url, androidId = crypto.randomBytes(16).toString("hex")) => {
  const deviceId = crypto.createHash("sha256")
    .update(`bypasstools:${androidId}`).digest("hex");

  const { sessionToken } = await fetch("https://bypass.tools/api/mobile/init", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ deviceId, platform: "android", appVersion: "1.0.0" })
  }).then(r => r.json());

  const data = await fetch("https://bypass.tools/api/mobile/bypass", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${sessionToken}`,
      "X-Device-ID": deviceId
    },
    body: JSON.stringify({ url, forceRefresh: false })
  }).then(r => r.json());

  if (!data.result) throw new Error(data.message || "Bypass failed");
  return data.result;
};

module.exports = bypass;
